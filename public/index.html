# Create a meeting
curl -X POST https://onepizza.io/api/meetings \
  -H "Content-Type: application/json" \
  -H "x-api-key: mk_your_key_here" \
  -d '{"title": "Team Standup"}'

# Response
{
  "meetingId": "team-standup-qk7",
  "adminToken": "eyJhb...",
  "joinUrl": "/join/team-standup-qk7",
  "createdAt": "2025-03-19T10:00:00Z"
}

# Download chat transcript
curl -O https://onepizza.io/api/meetings/team-standup-qk7/transcript/download \
  -H "x-api-key: mk_your_key_here"

# Upload a recording
curl -X POST https://onepizza.io/api/meetings/team-standup-qk7/recordings \
  -H "x-api-key: mk_your_key_here" \
  -F "recording=@meeting.webm"
