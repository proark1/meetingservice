Bot & Agent Quick Start
Connect an AI bot or agent to any meeting in 3 steps: create API key, create meeting via REST, join via Socket.IO.
import { io } from "socket.io-client";

// 1. Create meeting
const res = await fetch("/api/meetings", {
  method: "POST",
  headers: { "Content-Type": "application/json", "x-api-key": "mk_..." },
  body: JSON.stringify({ title: "Bot Meeting" })
});
const { meetingId, adminToken } = await res.json();

// 2. Connect & join
const socket = io("https://onepizza.io");
socket.emit("join-meeting", { meetingId, name: "My Bot", isAdmin: true, adminToken });

// 3. Listen & interact
socket.on("joined", ({ participantId }) => console.log("Joined:", participantId));
socket.on("chat:message", ({ name, text }) => console.log(`${name}: ${text}`));
socket.emit("chat:message", { text: "Hello from the bot!" });
